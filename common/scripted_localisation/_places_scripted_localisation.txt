defined_text = {
	name = places_toggle_loc
	text = { #Endonym-C, No MP
		trigger = {
			NOT = { has_global_flag = is_mp }
			has_global_flag = Endonyms-C_flag
		}
		localization_key = places_toggle_Endonym-C
	}
	text = { #Endonym-C, Yes MP
		trigger = {
			has_global_flag = is_mp
			has_global_flag = Endonyms-C_flag
		}
		localization_key = places_toggle_Endonym-C_mp
	}
	text = { #Endonym-S, No MP
		trigger = {
			NOT = { has_global_flag = is_mp }
			has_global_flag = Endonym-S_flag
		}
		localization_key = places_toggle_Endonym-S
	}
	text = { #Endonym-S, Yes MP
		trigger = {
			has_global_flag = is_mp
			has_global_flag = Endonym-S_flag
		}
		localization_key = places_toggle_Endonym-S_mp
	}
	text = { #Exonym
		trigger = {
			NOT = {
				has_global_flag = Endonyms-C_flag
				has_global_flag = Endonym-S_flag
			}
		}
		localization_key = places_toggle_Exonym
	}
}
